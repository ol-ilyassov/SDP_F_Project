package main
import "fmt"
type Person interface {
   SayHello()
}
type person struct {
   name        string
   age       int
}
func (p *person) SayHello()  {
   fmt.Printf("Hello, I am %v and %v years old", p.name,p.age)
}
type tiredPerson struct {
   name        string
   age       int
}
func (p *tiredPerson) SayHello()  {
   fmt.Println("Sorry I am sooo tired,I don't want speak!")
}
// Factory method with interface
func NewPerson(name string,age int) Person {
   if age > 100 {
      return &tiredPerson{name,age}
   }
   return &person{name, age}
}
func main()  {
   p := NewPerson("Bob",122)
   p.SayHello()
}
